852__attribute__((always_inline)) __STATIC_INLINE uint32_t __RBIT(uint32_t value)
853{
854 uint32_t result;
855
856#if ((defined (__ARM_ARCH_7M__ ) && (__ARM_ARCH_7M__ == 1)) || \
857 (defined (__ARM_ARCH_7EM__ ) && (__ARM_ARCH_7EM__ == 1)) || \
858 (defined (__ARM_ARCH_8M_MAIN__ ) && (__ARM_ARCH_8M_MAIN__ == 1)) )
859 __ASM volatile ("rbit %0, %1" : "=r" (result) : "r" (value) );
860#else
861 int32_t s = (4 /*sizeof(v)*/ * 8) - 1; /* extra shift needed at end */
862
863 result = value; /* r will be reversed bits of v; first get LSB of v */
864 for (value >>= 1U; value; value >>= 1U) {
865 result <<= 1U;
866 result |= value & 1U;
867 s--;
868 }
869 result <<= s; /* shift when v's highest bits are zero */
870#endif
871 return(result);
872}
